## 项目概述

本项目完整复刻了小米 SU7 官网的 3D 展示特效，包含以下功能：

| 功能 | 实现方式 |
|------|---------|
| LOADING 动画 | CSS 动画 + JS 过渡 |
| 进场动画 | GSAP timeline 编排 |
| 点击加速 | 射线检测 + GSAP 时间线 |
| Speed Lines | GLTF 模型 + 着色器 |
| 相机抖动 | Simplex Noise 算法 |
| HDR 环境切换 | 两个 HDR 动态混合 |
| Bloom 发光 | UnrealBloomPass 后处理 |
| 背景音乐 | Howler.js |

### 阶段四：资源加载

#### 4.1 资源管理器

```js
/**
 * AssetManager - 统一管理所有资源的加载
 * 支持：GLTF 模型、HDR 贴图、普通纹理
 */
class AssetManager {
  constructor() {
    this.items = {}    // 已加载的资源 { name: object }
    this.total = 0     // 总资源数
    this.loaded = 0    // 已加载数
    this.readyCallbacks = []
  }

  // 加载 GLTF 模型
  loadGLTF(name, path) {
    this.total++
    new GLTFLoader().load(path, (gltf) => {
      this.items[name] = gltf
      this.onLoaded()
    })
  }

  // 加载 HDR 环境贴图（用于环境反射）
  loadHDR(name, path) {
    this.total++
    new RGBELoader().load(path, (texture) => {
      texture.mapping = THREE.EquirectangularReflectionMapping
      this.items[name] = texture
      this.onLoaded()
    })
  }

  // 加载普通纹理
  loadTexture(name, path) {
    this.total++
    new THREE.TextureLoader().load(path, (texture) => {
      this.items[name] = texture
      this.onLoaded()
    })
  }

  onLoaded() {
    this.loaded++
    console.log(`[资源加载] ${this.loaded}/${this.total}`)
    if (this.loaded >= this.total) {
      this.readyCallbacks.forEach((cb) => cb())
    }
  }

  // 所有资源加载完成后执行
  onReady(cb) {
    if (this.loaded >= this.total) cb()
    else this.readyCallbacks.push(cb)
  }
}
```

### 阶段五：后处理（Bloom 发光）

#### 5.1 EffectComposer

```js
/**
 * Postprocessing - 后处理类
 * 使用 EffectComposer 添加 Bloom 发光效果
 */
class Postprocessing {
  constructor(experience) {
    this.composer = new EffectComposer(experience.renderer)

    // 1. 基础渲染通道（渲染场景到纹理）
    const renderPass = new RenderPass(experience.scene, experience.camera)
    this.composer.addPass(renderPass)

    // 2. Bloom 发光通道
    // Bloom 会让高亮区域（emissive 材质）周围产生光晕
    this.bloomPass = new UnrealBloomPass(
      new THREE.Vector2(width, height),
      0.8,   // strength - 发光强度，值越大光晕越亮
      0.4,   // radius - 发光半径
      0.85   // threshold - 阈值，低于此亮度的像素不发光
    )
    this.composer.addPass(this.bloomPass)
  }

  // 动态调整发光强度
  setIntensity(value) {
    this.bloomPass.strength = value
  }

  // 调整发光平滑度
  setLuminanceSmoothing(value) {
    this.bloomPass.luminanceMaterial.smoothing = value
  }
}
```

#### 5.2 Bloom 效果原理

```
原始画面 → 亮度提取 → 高斯模糊 → 与原图混合 → 发光效果

emissive 材质（车灯）的亮度 > threshold → 被检测为高亮
→ 模糊后叠加到原图 → 产生光晕效果
```

### 阶段六：动态环境贴图

#### 6.1 为什么需要动态环境贴图？

静态环境贴图只反映一种光照条件。但本项目需要模拟：
- **进场时**：黑暗 → 逐渐亮起
- **加速时**：正常 → 环境突然变暗

#### 6.2 实现原理

使用两个 HDR 环境贴图，通过 FBO 实时混合：

```js
/**
 * DynamicEnv - 动态环境贴图
 * 在两个 HDR 环境贴图之间平滑切换
 */
class DynamicEnv {
  constructor(experience, config = {}) {
    // FBO：Framebuffer Object，渲染到纹理
    this.fbo = new THREE.WebGLRenderTarget(1024, 1024)

    // 着色器：混合两个环境贴图
    this.material = new THREE.ShaderMaterial({
      uniforms: {
        uEnvmap1: { value: config.envmap1 || null },  // 夜晚 HDR
        uEnvmap2: { value: config.envmap2 || null },  // 白天 HDR
        uWeight: { value: 0 },    // 混合权重（GSAP 动画修改）
        uIntensity: { value: 1 }, // 整体强度
      },
      vertexShader: `
        varying vec2 vUv;
        void main() {
          vUv = uv;
          gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
        }
      `,
      fragmentShader: `
        uniform sampler2D uEnvmap1;
        uniform sampler2D uEnvmap2;
        uniform float uWeight;
        uniform float uIntensity;
        varying vec2 vUv;

        void main() {
          vec4 c1 = texture2D(uEnvmap1, vUv);
          vec4 c2 = texture2D(uEnvmap2, vUv);
          // mix(a, b, t) = a*(1-t) + b*t
          vec4 color = mix(c1, c2, uWeight) * uIntensity;
          gl_FragColor = color;
        }
      `,
    })

    // 全屏四边形
    this.quad = new THREE.Mesh(new THREE.PlaneGeometry(2, 2), this.material)

    // 暴露为纹理供场景使用
    this._envmap = this.fbo.texture
  }

  /** 每帧渲染到 FBO */
  update() {
    this.experience.renderer.setRenderTarget(this.fbo)
    this.experience.renderer.render(this.quad, this.experience.camera)
    this.experience.renderer.setRenderTarget(null)
  }
}
```

```js
/**
 * configureCar - 配置汽车模型
 * 遍历模型的所有网格，设置材质和贴图
 */
configureCar() {
  const scene = this.carModel.scene

  scene.traverse((child) => {
    if (!child.isMesh || !child.material) return
    const mat = child.material

    if (mat.isMeshStandardMaterial) {
      // 找到车身材质（通过颜色判断）
      if (!this.carBodyMat && mat.color?.getHex() !== 0x111111) {
        this.carBodyMat = mat
        mat.color.set(0x26d6e9)  // 小米 SU7 蓝色
        mat.envMapIntensity = 1  // 环境反射强度
      }

      // 所有网格都应用 AO 贴图
      mat.aoMap = this.am.items.carBodyAO
      mat.aoMapIntensity = 1.0
    }
  })
}
```

#### 7.2 展示厅配置

```js
/**
 * configureStartRoom - 配置展示厅
 * 包括地面材质（反射 + 贴图）和灯带材质（发光）
 */
configureStartRoom() {
  scene.traverse((child) => {
    if (!child.isMesh || !child.material) return
    const mat = child.material

    // 灯带材质（发光）
    if (mat.isMeshStandardMaterial && mat.emissive && mat.emissiveIntensity > 0) {
      mat.emissive.set(0x000000)     // 初始黑色
      mat.emissiveIntensity = 0       // 初始无光
      mat.toneMapped = false          // 不经过色调映射（保持发光）
      mat.transparent = true
      mat.alphaTest = 0.1
      this.startRoomLightMat = mat
    }

    // 地面材质（反射 + 贴图）
    if (mat.isMeshPhysicalMaterial) {
      mat.aoMap = this.am.items.startroomAO
      mat.lightMap = this.am.items.startroomLight
      mat.normalMap = this.am.items.floorNormal     // 凹凸感
      mat.roughnessMap = this.am.items.floorRoughness  // 粗糙度
      mat.envMapIntensity = 0
      mat.roughness = 0.05  // 很光滑 → 强反射
      mat.metalness = 0.9
      this.startRoomFloorMat = mat
    }
  })
}
```

#### 8.2 GSAP Timeline 进场动画

```js
/**
 * enter - 进场动画
 * 使用 GSAP timeline 编排多个动画同步进行
 */
enter() {
  params.disableInteract = true

  // ========== 时间线1：相机推进 ==========
  const t1 = gsap.timeline()
  this.tweens.push(t1)

  // 相机从 z=-11 推进到 z=-7
  t1.to(params.cameraPos, {
    x: 0, y: 0.8, z: -7,
    duration: 4,
    ease: 'power2.inOut',    // 慢入慢出，平滑自然
    onComplete: () => {
      params.isCameraMoving = false
      params.disableInteract = false
    },
  })

  // ========== 时间线2：灯光渐亮 ==========
  const t2 = gsap.timeline()
  this.tweens.push(t2)

  t2.to(params, {
    lightAlpha: 1,           // 颜色从黑变白
    lightIntensity: 1,     // 强度从0到1
    reflectIntensity: 25,   // 地面反射从0到25
    duration: 4,
    delay: 1,               // 延迟1秒后开始
    ease: 'power2.inOut',
    onUpdate: () => {
      // 颜色插值：黑 → 白
      lightColor.copy(blackColor).lerp(whiteColor, params.lightAlpha)
      this.startRoomLightMat.emissive.set(lightColor)
      this.startRoomLightMat.emissiveIntensity = params.lightIntensity
    },
  })

  // ========== 时间线3：环境贴图 ==========
  const t3 = gsap.timeline()
  this.tweens.push(t3)

  t3.to(params, {
    envIntensity: 1,
    duration: 4,
    delay: 0.5,
    onUpdate: () => {
      this.dynamicEnv.setIntensity(params.envIntensity)
    },
  }).to(params, {
    envWeight: 1,
    duration: 4,
    // '-=2.5' 表示比上一个动画提前 2.5 秒开始（重叠 1.5 秒）
    ease: 'power2.inOut',
    onUpdate: () => {
      this.dynamicEnv.setWeight(params.envWeight)
    },
  }, '-=2.5')
}
```

```js
/**
 * rush - 点击汽车后触发的加速模式
 */
rush() {
  this.clearAllTweens()  // 先清除所有现有动画

  // ========== 速度上升 ==========
  const t4 = gsap.timeline()
  t4.to(params, {
    speed: 4,            // 先到 4
    duration: 2,
    onComplete: () => {
      params.isRushing = true
      params.disableInteract = false
    },
  }).to(params, {
    speed: 10,           // 再到 10（最大速度）
    duration: 4,
  })

  // ========== 灯带变暗 ==========

  t5.to(params, {
    lightOpacity: 0,
    duration: 1,
    onUpdate: () => {
      this.startRoomLightMat.opacity = params.lightOpacity
    },
  })

  // ========== 地面变黑 ==========
  const t6 = gsap.timeline()
  t6.to(params, {
    floorLerpColor: 1,
    duration: 4,
    onUpdate: () => {
      const c = new THREE.Color(0xffffff).lerp(
        new THREE.Color(0x000000), params.floorLerpColor
      )
      this.startRoomFloorMat.color.set(c)
    },
  })

  // ========== 环境变暗 + FOV 增大 ==========
  setTimeout(() => {
    const t9 = gsap.timeline()
    t9.to(params, {
      carBodyEnvIntensity: 10,    // 车身反射暴增
      cameraShakeIntensity: 1,     // 相机开始抖动
      bloomLuminanceSmoothing: 0.4,
      bloomIntensity: 2,           // Bloom 加倍

        this.postprocessing.setLuminanceSmoothing(params.bloomLuminanceSmoothing)
        this.postprocessing.setIntensity(params.bloomIntensity)
      },
    })
  }, 1000)
}
```

### 阶段九：相机抖动

#### 9.1 Simplex Noise 算法

相机抖动使用 Simplex Noise 而不是纯随机，实现更「自然」的抖动：

```js
// Simplex 2D Noise 实现
function createNoise2D() {
  const p = []
  for (let i = 0; i < 256; i++) p[i] = Math.floor(Math.random() * 256)

  return function noise2D(x, y) {
    const X = Math.floor(x) & 255
    const Y = Math.floor(y) & 255
    x -= Math.floor(x)
    y -= Math.floor(y)
    const u = fade(x)
    const v = fade(y)
    const a = p[X] + Y, b = p[X + 1] + Y
    return lerp(v,
      lerp(u, grad(p[a & 255], x, y), grad(p[b & 255], x - 1, y)),
      lerp(u, grad(p[(a + 1) & 255], x, y - 1), grad(p[(b + 1) & 255], x - 1, y - 1))
    )
  }
}

function fade(t) { return t * t * t * (t * (t * 6 - 15) + 10) }
function lerp(t, a, b) { return a + t * (b - a) }
function grad(hash, x, y) {
  const h = hash & 15
  const u = h < 8 ? x : y
  const v = h < 4 ? y : h === 12 || h === 14 ? x : 0
  return ((h & 1) === 0 ? u : -u) + ((h & 2) === 0 ? v : -v)
}
```

#### 9.2 相机抖动实现

```js
class CameraShake {
  constructor(experience) {
    this.experience = experience
    this.intensity = 0
    this.tweenedOffset = new THREE.Vector3()
    this.noise2D = createNoise2D()
  }

  update() {
    if (this.intensity <= 0) return

    const t = performance.now() * 0.001

    // 用噪声生成 3D 偏移向量
    const offset = new THREE.Vector3(
      this.noise2D(t * 0.5, 0) * 2,
      this.noise2D(t * 0.5, 100) * 2,
      this.noise2D(t * 0.5, 200) * 2
    )

    // 平滑插值（避免抖动太突兀）
    offset.multiplyScalar(0.1 * this.intensity)
    this.tweenedOffset.lerp(offset, 0.1)

    // 添加到相机位置
    this.experience.camera.position.add(this.tweenedOffset)
  }
}
```

#### 10.1 射线检测

```js
setupInteraction() {
  const raycaster = new THREE.Raycaster()
  const mouse = new THREE.Vector2()

  window.addEventListener('click', (event) => {
    if (params.disableInteract) return

    // 归一化鼠标坐标 (-1 到 1)
    mouse.x = (event.clientX / window.innerWidth) * 2 - 1
    mouse.y = -(event.clientY / window.innerHeight) * 2 + 1

    // 从相机位置发射射线
    raycaster.setFromCamera(mouse, this.camera)

    // 检测是否与汽车模型相交
    const intersects = raycaster.intersectObject(this.carModel.scene, true)

    if (intersects.length > 0) {
      this.rush()  // 触发加速模式
    }
  })
}
```

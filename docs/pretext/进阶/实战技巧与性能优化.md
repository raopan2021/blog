# 实战技巧与性能优化

> 从文本自动高度到 Masonry 瀑布流，这些场景用 Pretext 怎么写

## 场景一：Textarea 自动高度

最经典的应用——输入框随内容自动伸缩：

```typescript
import { prepare, layout } from '@chenglou/pretext'

class AutoHeightTextarea {
  private prepared: PreparedText | null = null
  private textarea: HTMLTextAreaElement
  private font: string
  private lineHeight: number

  constructor(textarea: HTMLTextAreaElement) {
    this.textarea = textarea
    this.font = this.getFont()
    this.lineHeight = this.getLineHeight()

    // 等待字体加载完成
    document.fonts.ready.then(() => {
      this.update()
    })

    // 输入事件
    this.textarea.addEventListener('input', () => this.update())
    // 宽度变化（响应式）
    this.textarea.addEventListener('resize', () => this.update())
  }

  private getFont(): string {
    const style = getComputedStyle(this.textarea)
    // 从 CSS font shorthand 中提取
    const fontSize = style.fontSize
    const fontFamily = style.fontFamily
    return `${fontSize} ${fontFamily}`
  }

  private getLineHeight(): number {
    const style = getComputedStyle(this.textarea)
    const lh = style.lineHeight
    if (lh === 'normal') return parseFloat(this.font) * 1.5 // 默认约 1.5 倍
    return parseFloat(lh)
  }

  private update() {
    const value = this.textarea.value
    const width = this.textarea.clientWidth

    // 新文本或字体变了 → 重新 prepare
    if (!this.prepared) {
      this.prepared = prepare(value, this.font, { whiteSpace: 'pre-wrap' })
    }

    // 获取高度
    const { height } = layout(this.prepared, width, this.lineHeight)

    // 设置最小高度（与 CSS min-height 一致）
    this.textarea.style.height = Math.max(height, 80) + 'px'
  }
}

// 使用
const textarea = document.getElementById('auto-textarea')
new AutoHeightTextarea(textarea)
```

## 场景二：Canvas 文本渲染器

在 Canvas 上渲染多行文本（不依赖 DOM）：

```typescript
import { prepareWithSegments, layoutWithLines } from '@chenglou/pretext'

class CanvasTextRenderer {
  private ctx: CanvasRenderingContext2D
  private font: string
  private lineHeight: number
  private maxWidth: number

  constructor(canvas: HTMLCanvasElement) {
    this.ctx = canvas.getContext('2d')!
  }

  render(text: string, font: string, maxWidth: number, lineHeight: number) {
    this.font = font
    this.lineHeight = lineHeight
    this.maxWidth = maxWidth

    const prepared = prepareWithSegments(text, font)
    const { lines } = layoutWithLines(prepared, maxWidth, lineHeight)

    // 渲染每行
    this.ctx.font = font
    this.ctx.fillStyle = '#333'
    this.ctx.textBaseline = 'top'

    lines.forEach((line, i) => {
      const y = i * lineHeight
      this.ctx.fillText(line.text, 0, y)
    })

    // 返回总高度
    return lines.length * lineHeight
  }
}

// 使用
const canvas = document.getElementById('text-canvas') as HTMLCanvasElement
const renderer = new CanvasTextRenderer(canvas)

const height = renderer.render(
  '这是一段很长的文本...\n第二行...',
  '20px "PingFang SC", sans-serif',
  canvas.width,
  30
)

canvas.height = height // 自动调整 Canvas 高度
```

## 场景三：文本环绕浮动元素

文字绕着图片流动，是传统 DOM 很难优雅实现的场景：

```typescript
import { prepareWithSegments, layoutNextLine } from '@chenglou/pretext'

interface FloatBox {
  left: number
  top: number
  width: number
  height: number
}

function renderTextWithFloat(
  ctx: CanvasRenderingContext2D,
  text: string,
  font: string,
  lineHeight: number,
  columnWidth: number,
  floatBox: FloatBox
) {
  const prepared = prepareWithSegments(text, font)
  const floatBottom = floatBox.top + floatBox.height

  let cursor = { segmentIndex: 0, graphemeIndex: 0 }
  let y = 0

  ctx.font = font
  ctx.fillStyle = '#333'
  ctx.textBaseline = 'top'

  while (true) {
    // 判断当前行是否在浮动区域内
    const isInFloatZone = y < floatBottom
    const leftMargin = isInFloatZone ? floatBox.width + 10 : 0
    const availableWidth = columnWidth - leftMargin

    if (availableWidth <= 0) break // 剩余空间不足

    const line = layoutNextLine(prepared, cursor, availableWidth)
    if (line === null) break // 文本结束

    // 渲染
    const x = leftMargin
    ctx.fillText(line.text, x, y)

    // 移动光标
    cursor = line.end.cursor
    y += lineHeight
  }
}

// 使用
renderTextWithFloat(
  ctx,
  longArticleText,
  '16px Inter',
  24,
  canvas.width,
  { left: 20, top: 20, width: 150, height: 150 } // 左侧浮动图片
)
```

## 场景四：虚拟列表（真正的精准高度）

传统虚拟列表用估算高度，会出现滚动条跳动。Pretext 让"虚拟滚动 + 真实高度"成为可能：

```typescript
import { prepare, layout } from '@chenglou/pretext'

interface VirtualListOptions {
  container: HTMLElement
  items: { text: string; id: number }[]
  font: string
  estimatedHeight: number
  rowHeight: number
}

class TrueVirtualList {
  private container: HTMLElement
  private items: { text: string; id: number }[]
  private font: string
  private rowHeight: number
  private prepared: Map<number, PreparedText> = new Map()

  constructor(options: VirtualListOptions) {
    this.container = options.container
    this.items = options.items
    this.font = options.font
    this.rowHeight = options.rowHeight

    // 预处理所有 item
    document.fonts.ready.then(() => {
      this.items.forEach((item, i) => {
        this.prepared.set(item.id, prepare(item.text, this.font))
      })
      this.render()
    })

    this.container.addEventListener('scroll', () => this.render())
    window.addEventListener('resize', () => this.render())
  }

  // 计算每行真实高度
  getHeight(id: number, width: number): number {
    const prepared = this.prepared.get(id)
    if (!prepared) return this.rowHeight

    const { height } = layout(prepared, width, this.rowHeight)
    return height
  }

  render() {
    const scrollTop = this.container.scrollTop
    const viewportHeight = this.container.clientHeight
    const width = this.container.clientWidth

    // 计算总高度（关键！有了真实高度，滚动条才准确）
    let totalHeight = 0
    const offsets: number[] = []
    for (const item of this.items) {
      offsets.push(totalHeight)
      totalHeight += this.getHeight(item.id, width)
    }
    this.container.style.height = totalHeight + 'px'

    // 找出可视区域
    const startOffset = scrollTop
    const endOffset = scrollTop + viewportHeight

    let startIndex = 0
    let endIndex = this.items.length - 1

    for (let i = 0; i < offsets.length; i++) {
      if (offsets[i] + this.getHeight(this.items[i].id, width) > startOffset) {
        startIndex = Math.max(0, i - 1) // 多渲染一行作缓冲
        break
      }
    }

    for (let i = startIndex; i < offsets.length; i++) {
      if (offsets[i] > endOffset) {
        endIndex = Math.min(this.items.length - 1, i + 1)
        break
      }
    }

    // 渲染可视项（这里省略渲染逻辑，假设有 renderItem 方法）
    for (let i = startIndex; i <= endIndex; i++) {
      const item = this.items[i]
      const y = offsets[i]
      this.renderItem(item, 0, y, width)
    }
  }

  private renderItem(item: any, x: number, y: number, width: number) {
    // 渲染到 DOM 或 Canvas
    // ...
  }
}
```

## 场景五：防止布局抖动（CLS 优化）

Content Layout Shift（内容布局偏移）是 Core Web Vitals 指标之一。文本加载导致的抖动可以用 Pretext 消除：

```typescript
// 页面加载时，先用 Pretext 计算高度
// 在文本实际渲染前，就给容器设定正确的高度
// 这样文本加载后，容器高度已经固定，不会产生抖动

function initWithNoLayoutShift() {
  document.fonts.ready.then(() => {
    const prepared = prepare(articleText, '16px Inter/24px Inter')
    const { height } = layout(prepared, articleContainer.clientWidth, 24)

    // 先占位，避免抖动
    articleContainer.style.minHeight = height + 'px'

    // 然后再渲染文本
    articleContainer.innerHTML = articleText
  })
}
```

## 性能优化技巧

### 1. 批量 prepare

```typescript
// ❌ 逐个 prepare
for (const text of texts) {
  prepare(text, font) // 每次都触发 Canvas 操作
}

// ✅ 批量 prepare（在同一段代码中执行，共享字体加载开销）
const prepared = texts.map(text => prepare(text, font))
```

### 2. 宽度离散化缓存

如果宽度只有几种固定值（如响应式断点 375/768/1024），可以缓存：

```typescript
const widthCache = new Map<number, { height: number; lineCount: number }>()

function getHeight(prepared: PreparedText, width: number, lineHeight: number) {
  // 离散化宽度到最近的缓存点
  const bucket = Math.round(width / 10) * 10

  if (!widthCache.has(bucket)) {
    const result = layout(prepared, bucket, lineHeight)
    widthCache.set(bucket, result)
  }

  return widthCache.get(bucket)!
}
```

### 3. Web Worker 预处理

prepare() 可以在 Worker 中执行，不阻塞主线程：

```typescript
// worker.ts
self.onmessage = ({ data }) => {
  const { text, font, id } = data
  const prepared = prepare(text, font) // 在 Worker 中执行
  self.postMessage({ id, prepared })
}
```

## 总结

| 场景 | 核心 API |
|------|---------|
| Textarea 自动高度 | `prepare` + `layout` |
| Canvas 文本渲染 | `prepareWithSegments` + `layoutWithLines` |
| 文本环绕浮动 | `prepareWithSegments` + `layoutNextLine` |
| 精准虚拟列表 | `prepare` + `layout`（核心！计算真实高度） |
| CLS 优化 | `prepare` + `layout`（预计算占位） |
